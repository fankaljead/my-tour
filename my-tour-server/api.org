   - 获取所有用户
     - /user
     - method: get

   - 登录
     - /user/login
     - method: get
     - params: username, password
     - return:
      |   0 | 用户不存在 |
      | >=1 | 用户 id    |
      |  -1 | 密码错误   |
   - 用户注册
     - /user
     - method: post
     - params: username, password
     - return: user_id
   - 获取用户信息
     - /user/:uid
     - method: get
     - params: uid
     - return: user
   - 修改用户信息
     - /user/:uid
     - method: put
     - params: uid, user
     - return: 1 0
   - 删除用户
     - /user/:uid
     - method: delete
     - params: uid
     - return: 1 0
   - 退出登录
     - /user/logout
     - method: get
     - return 1 0
